#### 组件间传值

**父传子父组件**

```jsx
import React from 'react';
import Son from './son';

export default class Parent extends from React.Component {
	state = { info: '我是父组件需要传递的数据' }
  render() {
  	return(
    	<div className="parent">
        <h2>我是父组件</h2>
        <Son info={info}/>
      </div>
    )
  }
}
```

**父传子子组件**

```jsx
import React from 'react';

export default class Son extends React.Components {
	render() {
    const {info} =  this.props
		return(
      <h2>我是子组件</h2>
      <div>这是父组件传入的信息：{info}</div>
    )
  }
}

```



**子传父父组件(子组件通过回调函数获取父组件)**

```jsx
import React from 'react'
import Son from './son'

export default class Parent from React.Component {
	state = { info: '我是原始数据' }
  changeInfo = (info) => {
    this.setState({info});
  }
  render() {
  	const { info } = this.state
    return(
    	<div className='parent'>
      	<h2>我是父组件<h2/>
        <div>我目前的信息是：{info}</div>
        <Son changeInfo={info => {this.changeInfo(info)}}
      </div>
    )
  }
}

```

**子传父子组件**

```jsx
import React from 'react';

export default class Son extends React.Component {
	btnClick = () => {
  	const info = '我是子组件发送给父组件的数据';
    this.props.changeInfo(info);
  }
  render() {
  	return(
    	<div>
        <h2>我是子组件</h2>
        <div onClick={() => this.btnClick()}>点我发送信息</div>
      </div>
    )
  }
}

```
