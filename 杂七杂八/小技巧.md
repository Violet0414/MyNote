#### 复杂逻辑判断

可在此基础上做逻辑扩展，简化if，else，switch等大片逻辑判断代码块

```js
const one = {name: '张三', age: 18, hobby: '抽烟', };
const two = {name: '李四', age: 19, hobby: '喝酒', };
const three = {name: '王五', age: 20, hobby: '烫头', };
const defaults = {name: '狗蛋', age: 23, hobby: '睡觉', };

const switchList = new Map([
	[0, [one, () => { testOne()}]],
	[1, [two, () => { testTwo()}]],
	[2, [three, () => { testThree()}]],
	['default', [defaults, () => { defaultTest()}]],
])

function test(val = 'default') {
	let action = switchList.get(val)
	console.log('我叫' + action[0].name + ',今年' + action[0].age + '岁,喜欢' + action[0].hobby + '。');
	action[1].call(this);
}

function testOne() { console.log('别人给我就抽，不给我就要'); }
function testTwo() { console.log('开车不喝酒，喝酒不开车'); }
function testThree() { console.log('开业大酬宾，烫头八折起'); }
function defaultTest() { console.log('睡了晚安'); }

test(0);
test(1);
test(2);
test();

// 我叫张三,今年18岁,喜欢抽烟。
// 别人给我就抽，不给我就要
// 我叫李四,今年19岁,喜欢喝酒。
// 开车不喝酒，喝酒不开车
// 我叫王五,今年20岁,喜欢烫头。
// 开业大酬宾，烫头八折起
// 我叫狗蛋,今年23岁,喜欢睡觉。
// 睡了晚安
```
